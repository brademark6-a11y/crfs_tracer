    // ==================== LOAD FROM SUPABASE ONLY (NO LOCAL DEFAULTS) ====================
    async function loadFromSupabase() {
        if (!supabaseClient) {
            updateSyncStatus('offline', 'Offline');
            facilities = [];
            renderAll();
            showToast("⚠️ Offline mode - cannot load data");
            return;
        }
        
        updateSyncStatus('loading', 'Loading...');
        
        try {
            const { data, error } = await supabaseClient
                .from('farms')
                .select('*')
                .order('created_at', { ascending: false })
                .timeout(8000);
            
            if (error) throw error;
            
            if (data && data.length > 0) {
                facilities = data.map(item => ({
                    id: item.id,
                    name: item.name || 'Unnamed',
                    locationRaw: item.address || 'No address',
                    facilityType: item.farm_type || 'farm-poultry',
                    icon: getEmojiFromType(item.farm_type || 'farm-poultry'),
                    coordinates: [item.latitude, item.longitude],
                    ownerName: item.owner_name || 'Not specified',
                    contactNumber: item.contact_number || 'N/A',
                    scheduleTime: item.schedule || 'By appointment'
                }));
                updateSyncStatus('online', `${facilities.length} records`);
                showToast(`📦 Loaded ${facilities.length} facilities from cloud`);
            } else {
                facilities = [];
                updateSyncStatus('online', 'No data');
                showToast("📭 No facilities found. Add your first farm!");
            }
            
            renderAll();
            
        } catch (err) {
            console.error('Supabase load error:', err);
            facilities = [];
            updateSyncStatus('offline', 'Offline');
            renderAll();
            showToast("⚠️ Cannot connect to server. Using offline data if available.");
        }
    }

    }
    
    async function saveToSupabase(farmData) {
        if (!supabaseClient) return false;
        try {
            const { error } = await supabaseClient
                .from('farms')
                .insert([{
                    id: farmData.id,
                    name: farmData.name,
                    address: farmData.locationRaw,
                    farm_type: farmData.facilityType,
                    latitude: farmData.coordinates[0],
                    longitude: farmData.coordinates[1],
                    owner_name: farmData.ownerName,
                    contact_number: farmData.contactNumber,
                    schedule: farmData.scheduleTime,
                    created_at: new Date().toISOString()
                }]);
            if (error) throw error;
            return true;
        } catch (err) {
            console.warn('Save error:', err);
            return false;
        }
    }
    
    async function deleteFromSupabase(id) {
        if (!supabaseClient) return false;
        try {
            await supabaseClient.from('farms').delete().eq('id', id);
            return true;
        } catch (err) { return false; }
    }

            el.innerHTML = `<i class="fas fa-cloud-off"></i> ${text}`;
        }
    }

            ownerName: owner || 'Unspecified',
            contactNumber: contact || 'N/A',
            scheduleTime: sched || 'By appointment'
        };
        
        const saved = await saveToSupabase(newFacility);
        if (saved) {
            facilities.unshift(newFacility);
            renderAll();
            map.flyTo(coords, 15);
            if (tempMarker) map.removeLayer(tempMarker);
            selectedLatLng = null;
            
            document.getElementById('facilityName').value = '';
            document.getElementById('locationInput').value = '';
            document.getElementById('ownerName').value = '';
            document.getElementById('contactNumber').value = '';
            document.getElementById('scheduleTime').value = '';
            
            showToast(`✅ ${name} registered!`);
        } else {
            showToast("❌ Failed to save. Check connection.");
        }
    }
    
    async function deleteFacility(id) {
        if (confirm("Remove this facility?")) {
            const deleted = await deleteFromSupabase(id);
            if (deleted) {
                facilities = facilities.filter(f => f.id !== id);
                renderAll();
                if (currentRouteControl) { map.removeControl(currentRouteControl); currentRouteControl = null; }
                showToast("🗑️ Removed");
            } else {
                showToast("❌ Delete failed");
            }
        }
    }
    
    async function geocodeLocation(locationText) {
        try {
            const q = `${locationText}, Santa Maria, Bulacan, Philippines`;
            const res = await fetch(`https://nominatim.openstreetmap.org/search?format=json&q=${encodeURIComponent(q)}&limit=1`);
            const data = await res.json();
            if (data?.length) return { lat: parseFloat(data[0].lat), lng: parseFloat(data[0].lon) };
        } catch(e) {}
        return null;
    }

        if (currentTypeFilter === 'derma') return facilities.filter(f => f.facilityType === 'clinic-derma');

    }

    function drawRouteToAllFarms() {
        if (currentRouteControl) map.removeControl(currentRouteControl);
        const farmPoints = facilities.filter(f => f.facilityType.startsWith('farm'));
        if (!farmPoints.length) { showToast("Add farms first"); return; }
        const waypoints = [L.latLng(HQ_LOCATION.lat, HQ_LOCATION.lng), ...farmPoints.map(f => L.latLng(f.coordinates[0], f.coordinates[1]))];
        currentRouteControl = L.Routing.control({
            waypoints, routeWhileDragging: false, show: false,
            lineOptions: { styles: [{ color: '#E53E3E', weight: 6, opacity: 0.9 }], extendToWaypoints: true },
            createMarker: () => null, addWaypoints: false, fitSelectedRoutes: true
        }).addTo(map);
        currentRouteControl.on('routesfound', (e) => {
            const totalKm = (e.routes[0].summary.totalDistance / 1000).toFixed(1);
            showToast(`🚜 Route: ${farmPoints.length} farms | ${totalKm} km`);
            document.getElementById('routeStatus').innerHTML = `${totalKm}km`;
        });
        map.fitBounds(L.latLngBounds(waypoints), { padding: [45,45] });
    }

    function useMyLocation() { 
        if (!navigator.geolocation) return alert("Geolocation unavailable"); 
        navigator.geolocation.getCurrentPosition(pos => { 
            const { latitude: lat, longitude: lng } = pos.coords; 
            selectedLatLng = { lat, lng }; 
            if (tempMarker) map.removeLayer(tempMarker); 
            tempMarker = L.marker([lat, lng], { icon: L.divIcon({ html: '<div style="font-size:42px;">📍</div>', iconSize: [36,36] }) }).addTo(map).bindPopup("📍 Your GPS location").openPopup(); 
            map.setView([lat, lng], 16); 
            showToast("GPS locked"); 
        }, () => alert("Location access denied")); 
    }

    function calcDistance(lat1, lng1, lat2, lng2) { 
        const R = 6371; 
        const dLat = (lat2 - lat1) * Math.PI / 180; 
        const dLng = (lng2 - lng1) * Math.PI / 180; 
        const a = Math.sin(dLat/2)**2 + Math.cos(lat1 * Math.PI/180) * Math.cos(lat2 * Math.PI/180) * Math.sin(dLng/2)**2; 
        return R * 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a)); 
    }
